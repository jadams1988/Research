{
	Abundant=listcreate();
	for(i=1,28123,
		SumFactors=0;
		for(j=1,i-1,
			if(i%j==0,
				SumFactors+=j
			)
		);
		if(SumFactors>i,
			listput(Abundant,i)
		)
	);
	NumAbundant=length(Abundant);
	Sum2Abundant=vector(28123);
	for(i=1,28123,
		Sum2Abundant[i]=1
	);
	for(i=1,NumAbundant,
		for(j=i,NumAbundant,
			SUM=Abundant[i]+Abundant[j];
			if(SUM<=28123,
				Sum2Abundant[SUM]=0
			)
		)
	);
	TOTAL=0;
	for(i=1,28123,
		TOTAL+=(i*Sum2Abundant[i])
	);
	print(TOTAL)
}
